\section{Problema de los $N$-cuerpos}
\subsection{Ecuación}
\begin{frame}{Problema}

\onslide<+->\begin{block}{Sistema físico}

\begin{description}
\item[Espacio:] Espacio euclideano $d$ dimensional $d\geq 2$ ($\rr^d$).
\item[Objetos:] $N$-puntos de masas $m_1,m_2,\ldots,m_N$,
\item[Variables:] tiempo $t$, posiciones $\b{x}_i=\b{x}_i(t)\in \rr^d$, $i=1,\ldots,N$,
 \item[Fuerzas:] gravitacionales,
 \item[Leyes físicas:] Mecánica Newtoniana: Segunda Ley y Ley de gravitación universal.
 
\end{description}
 \end{block}
 \end{frame} 



\begin{frame}{Ecuación de los $N$-cuerpos}

\onslide<+->\begin{block}{Ecuaciones $N$-cuerpos}
\[\b{x}''_i(t)=G\sum_{j\neq i}m_j\frac{\b{x}_j-\b{x}_i}{\|\b{x}_j-\b{x}_i\|^3}.\]
$G$ constante gravitación universal, supondremos $G=1$.
\end{block}



 \end{frame}

\end{frame} 
 

 
\begin{frame}{Notaciones}
\[
  \begin{split}
    \b{x}&=(\b{x}_1,\ldots,\b{x}_N)\in\rr^{d N},\quad\text{vector configuración, $\b{x}$},\\
    r_{ij}&=\|\b{x}_i-\b{x}_j\|,\quad\text{distancias relativas},\\
    M&=
       \begin{pmatrix}
	  M_1           & 0_{d\times d} &\cdots & 0_{d\times d}\\
	  0_{d\times d} & M_2           &\cdots & 0_{d\times d}\\
	  \vdots        & \ddots         &      &\vdots        \\
	  0_{d\times d} & \cdots         &      & M_N           \\
       \end{pmatrix},\,
       M_j=\begin{pmatrix}
            m_j     &  \cdots &  0   \\
           \vdots  & \ddots  & \vdots  \\
              0     &     \cdots     & m_j\\                
           \end{pmatrix}\in\rr^{d\times d}
       \\
    U(\b{x})&=      \sum_{i<j}\frac{m_im_j}{r_{ij}}\quad\text{Potencial newtoniano}\\
    \Delta&=\{\b{x}_i=\b{x}_j| i,j=1,\ldots, n;\, i\neq j\},\\
    \rr^{dN}-\Delta &\quad\text{ Espacio de configuraciones }.\\ 
  \end{split}
\]

\end{frame} 


 
\begin{frame}{Notaciones}
\[
  \begin{split}
   \b{v}&=\b{x}'\quad\text{(Velocidades)}\\
   K(\b{v})&=\frac12\b{v}\cdot M\b{v}=\frac12\sum_{j=1}^N m_j\|\b{v}_j\|^2,\quad \text{(Energía  cinética)}\\
   \b{c}&=\frac{1}{m}\sum_{j=1}^Nm_j\b{x}_j,\quad\text{con }m:=\sum_{j=1}^Nm_j\quad\text{(Centro masas)}\\
     \b{p}&=\sum_{j=1}^Nm_j\b{v}_j,\quad\text{(momento total)}\\
     \omega_{kl}&=\sum_{j=1}^Nm_j \left(\b{x}_{jk}\b{v}_{jl}-\b{x}_{jl}\b{v}_{jk}\right),\,\, \text{(Momento angular total $\omega^t=-\omega$)}
  \end{split}
\]


\end{frame} 

\begin{frame}{Notaciones}
\[
  \begin{split}
  \b{x}-\b{c}&=(  \b{x}_1-\b{c},\ldots,  \b{x}_N-\b{c})\\
   I(\b{x})&= \frac12(\b{x}-\b{c})\cdot M(\b{x}-\b{c})\quad\text{(Momento de inercia)}\\
   &=\frac12\sum_{j=1}^N m_j\|\b{x}_j-\b{c}\|^2,\\
  \end{split}
\]


\begin{block}{Ecuación $N$-cuerpos}
 \[M\b{x}''(t)=\nabla U(\b{x}),\,\text{(Ecuación $N$-cuerpos)}\]
\end{block}



\end{frame} 


 
  \subsection{Simetrías}
 \begin{frame}{ \begin{tabular}{cc} \multirow{2}{*}{\includegraphics[scale=.5]{Noether.jpg}} & Simetrías e integrales primeras \\  &(Emmy Noether)\\\end{tabular}}
 \begin{itemize}
  \item<+-> \textbf{Simetría por traslaciones} ($\b{x}_j\rightarrow \b{x}_j+\b{v}_0t+\b{c}_0$, $\b{c}_0,\b{v}_0\in\rr^d$) $\Rightarrow$ \fbox{conservación $\b{p}=\b{c}'$}  $\Rightarrow$ movimiento rectilineo uniforme de $\b{c}$
    \item<+-> \textbf{Simetría por rotaciones} ($\b{x}_j\rightarrow Q\b{x}_j$, $Q\in O(d)$) $\Rightarrow$ \fbox{conservación $\omega$}
    \item<+-> La \textbf{energía total} $H:\underset{\text{Espacio fases}}{\underbrace{\left(\rr^d-\Delta\right)\times\rr^d}}\to  \rr$,
    \[H(\b{x},\b{v})=K(\b{v})-U(\b{x}),\]
    se conserva
  
 \end{itemize}



\end{frame} 

  \begin{frame}{Otras simetrías}
 \begin{itemize}
  \item<+-> $f:\rr^d\to \rr^m$ se denomina homogénea de grado $\alpha\in\rr$  si $f(t\b{x})=t^{\alpha}f(\b{x})$.
  \item<+-> Si $f$ es homogénea de grado $\alpha$ entonces $\nabla f(x)$ es homogénea de grado $\alpha-1$.
  \item<+-> \textbf{Teorema Euler:} Si $f$ es homogénea de grado $\alpha$ entonces $\nabla f(x)\cdot x=\alpha f(x)$.
  \item<+-> $U$ es homogénea de grado $-1$,  $\nabla U$ es homogénea de grado $-2$ e $I$  es homogénea de grado $2$.
  \item<+-> \textbf{Invariancia por traslaciones:} si $\b{c}_0\in\rr^d\Rightarrow U(\b{x}+\b{c}_0)=U(\b{x})$.
  \item<+-> \textbf{Invariancia por rotaciones:} si $Q\in O(d) \Rightarrow$ $ U(Q\b{x})=U(\b{x})$. Por consiguiente $\nabla U(Q\b{x})=Q\nabla U(\b{x})$.

 \end{itemize}



\end{frame}
 
 \section{Soluciones homográficas}
\subsection{Configuraciones Centrales}
\begin{frame}{Configuraciones centrales (CC)}
\onslide<+->\begin{block}{Definición} Una configuración de puntos masa, con masas $m_1,\ldots,m_N$ y posiciones $\b{x}_1,\ldots,\b{x}_N$ es central si existe $\lambda\in\rr$ tal que
\[\nabla U(\b{x})+\lambda M(\b{x}-\b{c})=\nabla U(\b{x})+\lambda \nabla I(\b{x}-\b{c})=0.\]
 
\end{block}
\onslide<+->\textbf{Observación:} En una configuración central el vector aceleración sobre cada cuerpo apunta hacia $\b{c}$  y es proporcional a la distancia a $\b{c}$.

\end{frame} 

\begin{frame}{Existencia de CC}
\begin{itemize}
 \item<+-> Cuando $N=2$ todas las configuraciones son centrales.
 \item<+-> Para masas arbitrarias la existencia es poco clara. 
 \item<+-> \textbf{Soluciones sencillas} Si todas las masas son iguales $m_1=m_2=\cdots=m_N$ podemos ponerlas como vértices de un polítopo, polígono regular del plano, poliedro regular (sólido platónico), etc. Claramente la configuración es central. Se puede agregar una masa arbitraria en el centro.
\item<+-> Notoriamente en el caso del regular $d$-simplex (triángulo equilátero, tetraedro, etc) tenemos CC independientemente de las masas.
\end{itemize}

\end{frame}





\begin{frame}{Soluciones homográficas}

\begin{block}{Definición}
 Una solución $\b{x}(t)$ de las ecuaciones de  los $N$-cuerpos es homográfica si, y sólo si, existe una función escalar $r:\rr\to (0,+\infty)$, una matricial $Q:\rr\to SO(d)$ y un vector fijo $\b{x}_0\in\rr^{dN}$ tal que
 \[\b{x}(t)-\b{c}(t)=r(t)Q(t)(\b{x}_0-\b{c}_0),\]
 donde $\b{c}_0$ y $\b{c}(t)$ son los centros de masas de $\b{x}_0$ y $\b{x}(t)$ respectivamente.
\end{block}




\end{frame} 


\subsection{Soluciones homotéticas}
\begin{frame}{Soluciones homotéticas}

\onslide<+->\begin{block}{Definicion}
 Si $Q=I$ (identidad $d\times d$) la solución se dice homotética.
\end{block}



\onslide<+-> Remplazando en la ecuación de los $N$ cuerpos
\[r''(t)M(\b{x}_0-\b{c}_0)=\frac{1}{r^2(t)}\nabla U(\b{x}_0)\]

Por consiguiente debe existir $\lambda\in\rr$
\[r''(t)=-\frac{\lambda}{r^2(t)}\quad\text{y}\quad  M(\b{x}_0-\b{c}_0)=-\lambda\nabla U(\b{x}_0).\]

$r(t)$ es una solución del problema de Kepler unidimensional y $\b{x}_0$ es una CC.


\end{frame}

\subsection{Movimientos planos}


\begin{frame}{Movimiento plano}


\onslide<+-> Suponiendo $d=2$ (movimiento plano):
\[Q(t)=\begin{pmatrix}
        \cos(\theta(t)) & \sen(\theta(t))\\
         -\sen(\theta(t)) & \cos(\theta(t))\\
       \end{pmatrix} \in SO(2)
 \]

\onslide<+-> Remplazando $\b{x}-\b{c}=rQ(\b{x}_0-\b{c}_0)$ en la ecuación 
\[r''-r\theta'^2=-\frac{\lambda}{r^2},\quad r\theta''+2r'\theta'=0\quad \text{y } -\lambda M(\b{x}_0-\b{c}_0)=\nabla U(\b{x}_0)\]

\onslide<+->  $r$, $\theta$ resuelven la ecuación (coordenadas polares) de la ecuación de los dos cuerpos. Los cuerpos describen elipses keplerianas homotéticas. 


\end{frame}

\subsection{Soluciones homográficas para $d>3$}

\begin{frame}{Soluciones homográficas para $d\geq 3$}
En $d=3$  la única solución no planar es el movimiento homotético \cite{wintner2014analytical,Moeckel1994,Moeckel2016}.

\onslide<+-> En $d>3$ la situación es más compleja \cite{albouy1997probleme,chenciner2011lagrange,chenciner2013angular},  ver \cite{Moeckel2016}.




\end{frame}

\begin{frame}{Soluciones homográficas $d>3$}

\onslide<+->\begin{block}{Teorema}
Sea $\b{x}_0$ una CC y $C=\langle \b{x}_{01}-\b{c}_0,\ldots,  \b{x}_{0N}-\b{c}_0\rangle$. Supongamos que existe $J\in\rr^{d\times d}$ antisimétrica que satisface $J^2|_C=-I|_C$. Si $r,\theta$ son solución del problema de Kepler plano, entonces existe 
\[\b{x}-\b{c}=r(t)Q(t)(\b{x}_0-\b{c}_0)\]
con
\[Q(t)=e^{\theta(t)J}.\]
es una solución homográfica. Toda solución homográfica no rígida  es de esta forma.

 
\end{block}



\end{frame}


\begin{frame}{Soluciones homográficas $d>3$}

\onslide<+->\begin{block}{Definición}
Si $A\in\rr^{d\times d}$ 
\[e^A=I+A+\frac{1}{2!}A^2+\frac{1}{3!}A^3+\cdots\]

 
\end{block}
\onslide<+-> \textbf{Ejemplo:} $J=\begin{pmatrix}
                      0 & -1\\
                      1 &0\\
                     \end{pmatrix}, e^{tJ}=\begin{pmatrix}
                      \cos(t) & -\sen(t)\\
                      \sen(t) &\cos(t)\\
                     \end{pmatrix}$


\onslide<+-> Si $A$ es antisimétrica $e^A$ es ortogonal:
\[\left[e^A\right]^te^A=e^{A^t}e^A=e^{-A}e^A=e^{-A+A}=I.\]
\end{frame}

\begin{frame}{Estructuras Hermitianas}

\begin{block}{Definición}
Una estructura Hermitiana sobre un subespacio $H$ viene dada por producto interno  sobre $H$ y una función lineal $J:H\to H$ que es antisimétrica respecto al producto interno con $J^2=-I$.
\end{block}
Si $J$ es la matríz del Teorema
\begin{itemize}
 \item<+-> $H=C+JC$ tendrá una estructura Hermitiana, es invariante y $J^2|_H=-I|_H$.
 \item<+-> Una transformación antisimétrica tiene rango par.
  \item<+->  Es una condición necesaria que $C$ este contenido en un espacio con dimensión par.
  \item<+-> A la vez es suficiente. 
\end{itemize}

 
 

\end{frame}


\begin{frame}{Estructuras Hermitianas}

\onslide<+->\begin{block}{Teorema}
Todo movimiento rígido ocurre en un espacio de dimensión par y existe una matriz de configuración $\b{x}_0$ (no necesariamente central) y una martriz antisimétrica $A\in\rr^{d\times d}$ tal que
\[\b{x}-\b{c}=Q(t)(\b{x}_0-\b{c}_0)\]
con
\[Q(t)=e^{\theta(t)A}.\] 
\end{block}

A su vez, para $d=4$,  existe $\b{x}_0$ no central tal que la fórmula anterior define un movimiento rígido.

\end{frame}


\section{Configuraciones Centrales}

\subsection{Invariancias}
\begin{frame}{Simetrías para CC}
$\b{x}$ es CC si existe $\lambda\in\rr$ 
\[\nabla U(\b{x})+\lambda \nabla I(\b{x}-\b{c})=0.\]
con
\[ U(\b{x})=      \sum_{i<j}\frac{m_im_j}{r_{ij}}\quad\text{e}\quad  I(\b{x})=\frac12\sum_{j=1}^N m_j\|\b{x}_j-\b{c}\|^2.\]


\begin{itemize}
\item<+-> Si $\b{x}$ es CC para $\lambda$ y $\b{c}_0\in\rr^d$ entonces $\b{x}+\b{c}_0$ es CC para el mismo $\lambda$.
 \item<+-> Si $\b{x}$ es CC para $\lambda$ y $Q\in SO(d)$ entonces $Q\b{x}$ es CC para el mismo $\lambda$.
 \item<+-> Si $\b{x}$ es CC para $\lambda$ y $\sigma\in \rr$ entoces $\sigma\b{x}$ es CC para $\sigma^{-3}\lambda$, 
\end{itemize}

\end{frame}


\subsection{Finitud de ER}

\begin{frame}{Finitud de equlibrios relativos}
\begin{block}{Definición}
 Dos CC $\b{x}$ e $\b{y}$ son equivalentes si existe $\sigma\in\rr$, $Q\in SO(d)$ y $\b{c}_0\in\rr^d$ tales que $\b{y}=\sigma Q\b{x}+\b{c}_0$.
\end{block}



\begin{block}{Problema de Chazy-Wintner-Smale, Steve Smale, ``Mathematical problems for the next century'', \cite{arnold2000mathematics}}
 \begin{quote} Existe un número finito de clases de equivalencias de CC.
   \end{quote}
\end{block}
\end{frame}


\begin{frame}{Finitud de equlibrios relativos ¿Qué sabemos? }
\begin{itemize}
 \item<+-> Para $N=2$ todo $\b{x}$ es $CC$, pero hay una sóla clase de equivalencia.
 \item<+-> Para $N=3$ hay 5 clases distintas, Euler-Lagrange.
 \item<+-> Para $N=4$. Dadas 4 masas $m_1,m_2,m_3,m_4$ el número de clases distintas está comprendido entre  32 y 8472  (\cite{hampton2006finiteness}).
 \item<+-> Para $N=5$ y $d=2$. En \cite{albouy2012finiteness} se obtuvo finitud para todas las masas excepto en una variedad de codimensión 2 en el espacio de masas.
\end{itemize}
\end{frame}


\subsection{Configuraciones de Euler, Lagrange y Moulton ($d=3$) }
\begin{frame}{CC y teoría puntos críticos \cite{smale1970topologyI,smale1970topologyII}}
\begin{itemize}
 \item Cambiando de coordenadas a baricéntricas podemos asumir $\b{c}=0$.
 \item Cambiando de escala podemos asumir $I(\b{x})=1$.
\end{itemize}

 Por la técnica de los multiplicadores de Lagrange, interpretamos
\[\nabla U(\b{x})+\lambda \nabla I(\b{x})=0.\]
como que una CC es un punto crítico del potencial con la restricción de que  $\b{x}\in S$ donde $S$  es el elipsoide
\[S=\{\b{x}| I(\b{x})=1\}.\]

\end{frame}


\begin{frame}{Soluciones Lagrangianas ($d=2$)}
\onslide<+->$U$ es función de las distancias relativas, $r_{ij}=\|\b{x}_i-\b{x}_j\|$. 

Veamos que $I$ también

\[
 \begin{split}
  \sum_{i,j=1}^N m_im_j\|\b{x}_i-\b{x}_j\|^2 &= 
  \sum_{i,j=1}^N m_im_j\|\b{x}_i\|^2\\
  &\,\,-
  2\sum_{i,j=1}^N m_im_j\b{x}_i\cdot\b{x}_j +
  \sum_{i,j=1}^N m_im_j\|\b{x}_j\|^2\\
  &=4mI
 \end{split}
\]
\onslide<+->Pongamos $\b{r}$ al vector de distancias relativas. Luego
\[\frac{\partial U}{\partial \b{x}}+\lambda \frac{\partial I}{\partial \b{x}}
 =\left(\frac{\partial U}{\partial \b{r}}+\lambda \frac{\partial I}{\partial \b{r}}\right) \frac{\partial \b{r}}{\partial \b{x}}\]



\end{frame}
\begin{frame}{Soluciones Lagrangianas ($d=2$)}


Queremos ver que  si los cuerpos no estan alineados
 \[0=\frac{\partial U}{\partial \b{x}}+\lambda \frac{\partial I}{\partial \b{x}}
 \Leftrightarrow 0=\frac{\partial U}{\partial \b{r}}+\lambda \frac{\partial I}{\partial \b{r}}.\]
 Pero
 \[
  \frac{\partial \b{r}}{\partial \b{x}}=
  \begin{pmatrix}
    \frac{\b{x}_1-\b{x}_2}{r_{12}}  & -  \frac{\b{x}_1-\b{x}_2}{r_{12}}  & 0\\
     \frac{\b{x}_1-\b{x}_3}{r_{13}} & 0                                 & -\frac{\b{x}_1-\b{x}_3}{r_{13}}\\
     0    & \frac{\b{x}_2-\b{x}_3}{r_{23}} & -  \frac{\b{x}_2-\b{x}_3}{r_{23}}
  \end{pmatrix}\in\rr^{3\times 6}
 \]
La afirmación sale de que  si los cuerpos no estan alineados $\b{x}_1-\b{x}_2$ y  $\b{x}_1-\b{x}_3$ son LI, lo mismo 
$\b{x}_1-\b{x}_2$ y  $\b{x}_2-\b{x}_3$ 
 


\end{frame}
\begin{frame}{Soluciones Lagrangianas ($d=2$)}
Así tenemos que resolver
 \[0=-\frac{m_im_j}{r_{ij}^2}+\lambda m_im_jr_{ij}\]
Entonces
\[r_{ij}=\lambda^{-1/3}.\]
La configuración es un triángulo equilátero.
 


\end{frame}

\begin{frame}{Configuraciones de Moulton (colineales)}
 \onslide<+->\begin{block}{Definición}
Una CC se denomina colineal si todos los cuerpos estan sobre una línea recta.

 \end{block}

\end{frame}

\begin{frame}{Demostración Teorema de Moulton }
 \onslide<+-> Usando un conveniente sistema de coordenadas podemos suponer las posiciones en $\rr$,   $x_1,x_2,x_3\in\rr$. Escribamos $\b{x}=(x_1,x_2,x_3)$. 
 
 \onslide<+-> Sean
 \[\begin{split}
  H&=\{\b{x}| m_1x_1+m_2x_2+m_3x_3=0\}\quad\text{(hiperplano)}\\
  S&=\{\b{x}| m_1x_1^2+m_2x_2^2+m_3x_3^2=0\}\quad\text{(elipsoide)}\\
  \Delta&=\{x_i=x_j|i\neq j\}\quad\text{(conjunto de colisiones)}\\
  \end{split}
 \]




 
 \onslide<+-> Se observa que hay que encontrar un punto crítico de $U$ en el conjunto 
 \[S\cap H\setminus \Delta.\]
 
 \onslide<+-> $S\cap H$ es un elipsoide de una dimensión menor, en el caso $N=3$ es una elipse
\end{frame}
